How it works

  1. Deploy the agent (Docker/K8s/binary) or run agent‑less collector.
  2. Point your apps or exporters to the agent, or list scrape targets.
  3. Query metrics in Grafana. Ship to your existing backends.
# Minimal collector config
receivers:
  otlp:
    protocols:
      grpc: { endpoint: 0.0.0.0:4317 }
      http: { endpoint: 0.0.0.0:4318 }
  prometheus_simple:
    endpoint: "sample-app:8000"
    metrics_path: "/metrics"
processors:
  batch: {}
exporters:
  prometheus:
    endpoint: "0.0.0.0:${PORT}"
service:
  pipelines:
    metrics:
      receivers: [otlp, prometheus_simple]
      processors: [batch]
      exporters: [prometheus]
